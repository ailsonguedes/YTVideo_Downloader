import os
from flask import Flask, request, send_from_directory
from pytube import YouTube
from moviepy.editor import VideoFileClip

app = Flask(__name__)

@app.route("/", methods=["GET", "POST"])
def index():
    if request.method == "POST":
        url = request.form["url"]
        
        try:
            yt = YouTube(url)
            stream = yt.streams.get_highest_resolution()
            
            video_title = yt.title
            original_filename = f"{video_title}.mp4"
            converted_filename = f"{video_title}_converted.mp4"

            # Baixar o vídeo em seu formato original
            stream.download(output_path="downloads", filename=original_filename)
            
            # Converter o vídeo para MP4
            original_path = os.path.join("downloads", original_filename)
            converted_path = os.path.join("downloads", converted_filename)
            clip = VideoFileClip(original_path)
            clip.write_videofile(converted_path, codec="libx264")
            
            # Oferecer o download do arquivo convertido
            return send_from_directory("downloads", converted_filename, as_attachment=True)

        except Exception as e:
            return f"Erro: {str(e)}"

    return """
    <form method="POST">
        URL do vídeo do YouTube: <input type="text" name="url">
        <input type="submit" value="Baixar e Converter">
    </form>
    """

if __name__ == "__main__":
    if not os.path.exists("downloads"):
        os.makedirs("downloads")
    app.run(debug=True)
